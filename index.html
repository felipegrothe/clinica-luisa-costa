aprovação antes de atualizar estoque por NF'); -- ============================================================ -- VIEWS ÚTEIS -- ============================================================ -- Estoque atual por item (soma de todos os lotes aprovados) create or replace view public.estoque_atual as select i.id, i.nome, i.unidade, i.amp_por_caixa, i.estoque_min, i.estoque_ideal, s.nome as setor, coalesce(sum(l.disponivel_base) filter (where l.status = 'aprovado' and l.validade >= current_date), 0) as disponivel, coalesce(sum(l.quantidade_base) filter (where l.status = 'aprovado'), 0) as total_entrada, min(l.validade) filter (where l.status = 'aprovado' and l.disponivel_base > 0) as proxima_validade, case when coalesce(sum(l.disponivel_base) filter (where l.status = 'aprovado' and l.validade >= current_date), 0) <= i.estoque_min then 'critico' when coalesce(sum(l.disponivel_base) filter (where l.status = 'aprovado' and l.validade >= current_date), 0) <= i.estoque_ideal * 0.5 then 'baixo' else 'ok' end as status_estoque, case when min(l.validade) filter (where l.status = 'aprovado' and l.disponivel_base > 0) < current_date then 'vencido' when min(l.validade) filter (where l.status = 'aprovado' and l.disponivel_base > 0) <= current_date + interval '30 days' then 'critico' when min(l.validade) filter (where l.status = 'aprovado' and l.disponivel_base > 0) <= current_date + interval '90 days' then 'atencao' else 'ok' end as status_validade, -- preço médio histórico por caixa case when coalesce(sum(l.quantidade_cx) filter (where l.status = 'aprovado'), 0) > 0 then round(sum(l.preco_total) filter (where l.status = 'aprovado') / sum(l.quantidade_cx) filter (where l.status = 'aprovado'), 2) else 0 end as preco_medio_cx, -- último preço pago (select l2.preco_por_cx from public.lotes l2 where l2.item_id = i.id and l2.status = 'aprovado' order by l2.criado_em desc limit 1) as ultimo_preco_cx, -- custo estimado do estoque disponível case when coalesce(sum(l.quantidade_cx) filter (where l.status = 'aprovado'), 0) > 0 then round( coalesce(sum(l.disponivel_base) filter (where l.status = 'aprovado'), 0)::numeric / i.amp_por_caixa * round(sum(l.preco_total) filter (where l.status = 'aprovado') / sum(l.quantidade_cx) filter (where l.status = 'aprovado'), 2) , 2) else 0 end as custo_estoque, i.ativo from public.itens i join public.setores s on s.id = i.setor_id left join public.lotes l on l.item_id = i.id where i.ativo = true group by i.id, i.nome, i.unidade, i.amp_por_caixa, i.estoque_min, i.estoque_ideal, s.nome; -- Lotes próximos do vencimento ou vencidos create or replace view public.alertas_validade as select l.id as lote_id, i.nome as item_nome, s.nome as setor, l.numero_lote, l.validade, l.disponivel_base, (l.validade - current_date) as dias_para_vencer, case when l.validade < current_date then 'vencido' when l.validade <= current_date + interval '30 days' then 'critico' when l.validade <= current_date + interval '90 days' then 'atencao' else 'ok' end as status from public.lotes l join public.itens i on i.id = l.item_id join public.setores s on s.id = i.setor_id where l.status = 'aprovado' and l.disponivel_base > 0 and l.validade <= current_date + interval '90 days' order by l.validade asc; -- Dashboard executivo create or replace view public.dashboard as select (select count(*) from public.itens where ativo = true) as total_itens, (select count(*) from public.estoque_atual where status_validade = 'vencido') as itens_vencidos, (select count(*) from public.estoque_atual where status_validade = 'critico') as itens_vencendo, (select count(*) from public.estoque_atual where status_estoque = 'critico') as itens_estoque_minimo, (select coalesce(sum(custo_estoque),0) from public.estoque_atual) as valor_total_estoque, (select count(*) from public.lotes where status = 'pendente') as nfs_pendentes_aprovacao, (select count(*) from public.alertas_preco where visualizado = false) as alertas_preco_novos, (select coalesce(sum(ampolas_disp),0) from public.tirzepatida_frascos where status = 'ativo') as tirz_ampolas_total, (select coalesce(sum(preco_pago),0) from public.tirzepatida_frascos) as tirz_total_investido; -- Comparativo de fornecedores por item create or replace view public.comparativo_fornecedores as select i.nome as item_nome, f.nome as fornecedor, count(l.id) as total_compras, round(avg(l.preco_por_cx), 2) as preco_medio_cx, min(l.preco_por_cx) as menor_preco, max(l.preco_por_cx) as maior_preco, max(l.criado_em) as ultima_compra from public.lotes l join public.itens i on i.id = l.item_id join public.fornecedores f on f.id = l.fornecedor_id where l.status = 'aprovado' and l.preco_por_cx > 0 group by i.nome, f.nome order by i.nome, preco_medio_cx asc; -- ============================================================ -- ROW LEVEL SECURITY (RLS) -- Controle de acesso por nível de usuário --
