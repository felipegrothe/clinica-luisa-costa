create table if not exists sessoes ( id uuid primary key default gen_random_uuid(), created_at timestamptz default now(), protocolo_id uuid not null references protocolos(id) on delete cascade, semana_num integer not null check (semana_num between 1 and 12), data_sessao date, hora text, enfermeiro text, obs text, intercorrencias text, realizada boolean default false, itens jsonb default '{}', assinatura text, unique(protocolo_id, semana_num) ); alter table sessoes enable row level security; create policy "auth users full access" on sessoes for all to authenticated using (true) with check (true); create index if not exists idx_sessoes_protocolo on sessoes(protocolo_id);
